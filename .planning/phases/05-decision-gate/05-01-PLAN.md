**Step 2 — Spot-confirm telemetry status for direct deps (CLAUDE.md §Audit obligatoire item 2):**

For each direct dep, run a fast grep against the pub-cache source. If ~/.pub-cache is on the host, run:

```bash
PUB_CACHE_HOSTED="${PUB_CACHE:-$HOME/.pub-cache}/hosted/pub.dev"
for pkg in cupertino_icons flutter_map vector_map_tiles vector_map_tiles_pmtiles vector_tile_renderer pmtiles latlong2 permission_handler geolocator path_provider go_router logging path share_plus; do
  ver=$(grep -E "^  $pkg:" pubspec.lock -A 3 | grep -E "version: " | head -1 | sed -E 's/.*version: "([^"]+)".*/\1/')
  dir="$PUB_CACHE_HOSTED/${pkg}-${ver}"
  if [ -d "$dir" ]; then
    matches=$(grep -RIE "Firebase|Crashlytics|Sentry|Mixpanel|Amplitude|Segment|AppsFlyer|Adjust|Bugsnag" "$dir" 2>/dev/null | wc -l)
    echo "$pkg-$ver: $matches telemetry hits"
  else
    echo "$pkg-$ver: pub-cache dir missing — skip (Phase 1 audit holds)"
  fi
done
```
