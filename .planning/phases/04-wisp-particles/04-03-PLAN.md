Class skeleton:
```dart
class WispParticleSystem {
  WispParticleSystem({
    int maxCount = kMirkPocWispMaxCount,
    int rngSeed = 1337,
    Stopwatch? wallClock,
  })  : _maxCount = maxCount,
        _rng = math.Random(rngSeed),
        _wallClock = wallClock ?? (Stopwatch()..start());

  final int _maxCount;
  final math.Random _rng;
  final Stopwatch _wallClock;
  final List<WispParticle> _wisps = <WispParticle>[];
  final Set<String> _alreadySpawnedDiscIds = <String>{};
  int _spawnCounterSinceLastRollup = 0;

  Iterable<WispParticle> get wisps => _wisps;
  int get activeCount => _wisps.length;

  void spawnAtNewDisc({required String discId, required RevealDisc disc}) {
    // Idempotency guard FIRST.
    if (_alreadySpawnedDiscIds.contains(discId)) return;
    _alreadySpawnedDiscIds.add(discId);

    // WISP-03 warmup gate AFTER recording. Disc ID stays in the
    // already-seen set so post-warmup re-calls hit the idempotency
    // guard and don't fire either.
    if (_wallClock.elapsedMilliseconds < (kMirkPocWispWarmUpSeconds * 1000).round()) {
      return;
    }

    final radiusMeters = disc.radiusMeters;
    final circumferenceMeters = 2 * math.pi * radiusMeters;
    final sampleCount = (circumferenceMeters / kMirkPocWispMetersPerWisp).round();

    for (var i = 0; i < sampleCount; i++) {
      final theta = (i / sampleCount) * 2 * math.pi;
      final dLat = radiusMeters * math.sin(theta) / kMetersPerDegreeLat;
      final dLon = radiusMeters * math.cos(theta) /
          (kMetersPerDegreeLat * math.cos(disc.lat * math.pi / 180.0));
      final spawnLatLng = LatLng(disc.lat + dLat, disc.lon + dLon);
      final unitDirection = Offset(math.cos(theta), math.sin(theta));
      _spawnAtPosition(position: spawnLatLng, direction: unitDirection, atLatForLonScale: disc.lat);
      _spawnCounterSinceLastRollup += 1;
    }
  }

  void _spawnAtPosition({
    required LatLng position,
    required Offset direction,
    required double atLatForLonScale,
  }) {
    // ±0.5 m jitter to break visual lockstep on multi-particle bursts.
    final jitterMetersX = (_rng.nextDouble() - 0.5) * 1.0;  // ±0.5 m
    final jitterMetersY = (_rng.nextDouble() - 0.5) * 1.0;
    final jitterDLat = jitterMetersY / kMetersPerDegreeLat;
    final jitterDLon = jitterMetersX /
        (kMetersPerDegreeLat * math.cos(atLatForLonScale * math.pi / 180.0));
    final jitteredPosition = LatLng(position.latitude + jitterDLat, position.longitude + jitterDLon);

    // Speed factor breaks lockstep at the magnitude axis.
    final speedFactor = 0.8 + _rng.nextDouble() * 0.4;
    final velocity = Offset(
      direction.dx * kMirkPocWispDriftMetersPerSecond * speedFactor,
      direction.dy * kMirkPocWispDriftMetersPerSecond * speedFactor,
    );
    _wisps.add(
      WispParticle(
        position: jitteredPosition,
        velocityMetersPerSecond: velocity,
        life: kMirkPocWispLifeSeconds,
        maxLife: kMirkPocWispLifeSeconds,
      ),
    );
    _enforceCap();
  }

  void _enforceCap() {
    if (_wisps.length <= _maxCount) return;
    _wisps.sort((a, b) => b.life.compareTo(a.life));
    _wisps.removeRange(_maxCount, _wisps.length);
  }

  int _lastAdvanceMicros = 0;

  /// Production convenience — wraps [advance] with dt computed from
  /// the wall-clock Stopwatch. Used by [_FogPainter._renderWisps]
  /// (Plan 04-04). Tests prefer the underlying [advance] with explicit
  /// dt for determinism.
  ///
  /// First-call guard: if `_lastAdvanceMicros == 0` we just record the
  /// current wall-clock and skip integration (no dt to integrate over).
  /// Subsequent calls: dt = (current - last) / 1e6, clamped to
  /// [0, kMirkPocWispMaxDtSeconds] (= 0.1 s) so a paused-then-resumed
  /// painter doesn't snap-integrate over multiple seconds.
  void advanceFromWallClock(Stopwatch wallClock) {
    final currentMicros = wallClock.elapsedMicroseconds;
    if (_lastAdvanceMicros == 0) {
      _lastAdvanceMicros = currentMicros;
      return;
    }
    final dtSeconds = ((currentMicros - _lastAdvanceMicros) / 1e6)
        .clamp(0.0, kMirkPocWispMaxDtSeconds);
    _lastAdvanceMicros = currentMicros;
    if (dtSeconds > 0.0) {
      advance(dtSeconds);
    }
  }

  void advance(double dt) {
    // Iterate in reverse so removeAt doesn't shift indices.
    // Documented exception to CLAUDE.md "ne jamais muter une collection
    // pendant son itération" — single-item removal at N≤200, the
    // collect-then-remove pattern would allocate a List per advance call.
    for (var i = _wisps.length - 1; i >= 0; i--) {
      final w = _wisps[i];

      // Curl-noise input projection: LatLng → local-tangent-plane-meters-
      // equivalent. The kMelunCenter anchor provides a stable origin so
      // the noise field is deterministic at the same world position
      // regardless of the wisp's individual age. Magnitude factor 1000.0
      // brings the input into a curl-noise-friendly scale (compare to
      // donor's `position * 0.005` in screen-px basis at zoom 13 where
      // 1 raw px ≈ 9.55 m → 0.005 / 9.55 ≈ 5.2e-4 per metre; we use
      // 1000 / kMetersPerDegreeLat ≈ 8.98e-3 per degree = 8.98e-3 ×
      // 9.55e-6 deg/m ≈ 8.6e-8 per metre. Both produce visually
      // similar "organic drift" character; documented per CONTEXT
      // §Claude's Discretion).
      final curlInput = Offset(

      if (w.isDead) {
        _wisps.removeAt(i);
      }
    }
```
