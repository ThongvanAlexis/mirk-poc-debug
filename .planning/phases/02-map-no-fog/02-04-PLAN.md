   import 'package:flutter/material.dart';
   import 'package:flutter_map/flutter_map.dart';
   import 'package:geolocator/geolocator.dart';
   import 'package:latlong2/latlong.dart';

   import 'package:mirk_poc_debug/config/constants.dart';
   import 'package:mirk_poc_debug/l10n/app_localizations.dart';

   /// Recenter FAB (LOC-04 + LOC-05).
   ///
   /// Tap → animates the camera to `lastFix.latLng` at zoom 15 over 500 ms
   /// (Curves.easeInOut). Disabled when `lastFix == null`.
   ///
   /// Repeat-tap cancellation: a second tap mid-animation disposes the
   /// in-flight controller, snapshots the just-interpolated camera state,
   /// and starts a fresh 500 ms tween toward the latest `lastFix`. No
   /// flicker (CONTEXT-locked behaviour).
   ///
   /// Hand-rolled (~25 LOC core) per RESEARCH §LOC-04 — under the CONTEXT
   /// 30-LOC threshold; no `flutter_map_animations` dependency added.
   class RecenterFab extends StatefulWidget {
     const RecenterFab({super.key, required this.mapController, required this.lastFix});

     final MapController mapController;
     final Position? lastFix;

     @override
     State<RecenterFab> createState() => _RecenterFabState();
   }

   class _RecenterFabState extends State<RecenterFab> with TickerProviderStateMixin {
     AnimationController? _controller;

     void _onPressed() {
       final fix = widget.lastFix;
       if (fix == null) return;                            // belt-and-braces; FAB is also disabled in build
       _controller?.dispose();
       final cam = widget.mapController.camera;
       final fromLat = cam.center.latitude;
       final fromLon = cam.center.longitude;
       final fromZoom = cam.zoom;
       final toLat = fix.latitude;
       final toLon = fix.longitude;
       const toZoom = kPocRecenterZoom;
       final c = AnimationController(
         vsync: this,
         duration: const Duration(milliseconds: kPocRecenterAnimationMs),
       );
       final t = CurvedAnimation(parent: c, curve: Curves.easeInOut);
       t.addListener(() {
         final v = t.value;
         widget.mapController.move(
           LatLng(fromLat + (toLat - fromLat) * v, fromLon + (toLon - fromLon) * v),
           fromZoom + (toZoom - fromZoom) * v,
         );
       });
       _controller = c;
       c.forward();
     }

     @override
     void dispose() {
       _controller?.dispose();
       super.dispose();
     }

     @override
     Widget build(BuildContext context) {
       final l10n = AppLocalizations.of(context)!;
       return FloatingActionButton(
         tooltip: l10n.recenterTooltip,
         onPressed: widget.lastFix == null ? null : _onPressed,        // LOC-05 disabled state
         child: const Icon(Icons.my_location),
       );
     }
   }
   ```

   **Notes on the implementation:**
   - `TickerProviderStateMixin` (not `SingleTickerProviderStateMixin`) so
     repeat-tap creates a fresh controller each time without "Ticker
     attached to multiple states" assertions.
   - The closure captures `from*` values at tap time; if a third tap
     fires after a 200 ms second-tap snapshot, the third tap's snapshot
     is taken from the just-interpolated state via
     `widget.mapController.camera` — the previous controller's listener
     stops firing the moment we call `_controller?.dispose()`.
   - No need to call `_controller = null` after dispose; the next
     `_onPressed` overwrites the field.

   // Copyright (c) 2026 THONGVAN Alexis
   // Licensed under the Good Old Software License v1.0
   // See LICENSE file for details

   import 'dart:async';
   import 'dart:math' as math;

   import 'package:flutter/material.dart';
   import 'package:flutter_map/flutter_map.dart';

   import 'package:mirk_poc_debug/config/constants.dart';
   import 'package:mirk_poc_debug/l10n/app_localizations.dart';

   /// Always-visible compass icon (CONTEXT §Map camera bounds & gestures).
   ///
   /// Top-right position is set by the parent `MapScreen` (Plan 02-05);
   /// this widget owns its bearing-stream subscription and snap-to-north
   /// tween only.
   ///
   /// Subscribes to `mapController.mapEventStream` filtered to
   /// `MapEventRotate` and calls `setState` on each event so the icon
   /// glyph (Transform.rotate) stays pointing at world-north.
   ///
   /// On tap: 250 ms `Curves.easeInOut` tween calls `mapController.rotate`
   /// per frame ending at 0°. Uses **shortest-path math** so a bearing of
   /// 350° rotates +10° to north, not -350° (RESEARCH Open Question #2).
   class MapCompass extends StatefulWidget {
     const MapCompass({super.key, required this.mapController});

     final MapController mapController;

     @override
     State<MapCompass> createState() => _MapCompassState();
   }

   class _MapCompassState extends State<MapCompass> with TickerProviderStateMixin {
     StreamSubscription<MapEvent>? _eventSub;
     AnimationController? _controller;
     double _bearingDegrees = 0;

     @override
     void initState() {
       super.initState();
       _bearingDegrees = widget.mapController.camera.rotation;
       _eventSub = widget.mapController.mapEventStream.listen((MapEvent event) {
         if (event is MapEventRotate) {
           if (!mounted) return;
           setState(() => _bearingDegrees = event.currentRotation);
         }
       });
     }

     @override
     void dispose() {
       _eventSub?.cancel();
       _controller?.dispose();
       super.dispose();
     }

     /// Shortest-path delta to north on the circle. Returns a value in [-180, 180].
     /// Bearing 350° → delta +10° (forward to 360°/0°), not -350°.
     static double _shortestPathToNorth(double currentDegrees) {
       final normalized = ((-currentDegrees + 540) % 360) - 180;
       return normalized;
     }

     void _onPressed() {
       _controller?.dispose();
       final from = _bearingDegrees;
       final delta = _shortestPathToNorth(from);                // [-180, 180], target = from + delta = 0 mod 360
       if (delta == 0) return;                                   // already at north
       final c = AnimationController(
         vsync: this,
         duration: const Duration(milliseconds: kPocCompassAnimationMs),
       );
       final t = CurvedAnimation(parent: c, curve: Curves.easeInOut);
       t.addListener(() {
         final v = t.value;
         widget.mapController.rotate(from + delta * v);
       });
       _controller = c;
       c.forward();
     }

     @override
     Widget build(BuildContext context) {
       final l10n = AppLocalizations.of(context)!;
       final radians = _bearingDegrees * math.pi / 180.0;
       return IconButton(
         tooltip: l10n.compassTooltip,
         onPressed: _onPressed,
         icon: Transform.rotate(
           angle: -radians,                                       // glyph rotates opposite to camera so needle points to world-north
           child: const Icon(Icons.explore),
         ),
       );
     }
   }
   ```
