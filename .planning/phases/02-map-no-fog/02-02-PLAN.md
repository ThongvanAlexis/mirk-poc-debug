   // Copyright (c) 2026 THONGVAN Alexis

   import 'dart:async';
   import 'dart:io';

   import 'package:flutter/foundation.dart' show visibleForTesting;
   import 'package:flutter/services.dart' show rootBundle;
   import 'package:logging/logging.dart';
   import 'package:path/path.dart' as p;
   import 'package:path_provider/path_provider.dart';

   import 'package:mirk_poc_debug/config/constants.dart';

   /// Copies the bundled PMTiles archive from `rootBundle` to
   /// `<getApplicationSupportDirectory()>/maps/` exactly once per install
   /// (idempotent on subsequent launches via size-parity check).
   ///
   /// Why Support and not Documents: iOS Documents is iCloud-backed by
   /// default; a 4 MB binary blob there eats the user's iCloud budget on
   /// every launch (Pitfall 4 — verified against Apple's path_provider docs).
   class PmtilesAssetCopier {
     static final Logger _log = Logger('infrastructure.pmtiles');

     /// Test seam — when non-null, [ensureCopied] returns this future
     /// instead of touching the real filesystem. Mirrors the Phase 1
     /// PermissionHandlerPlatform.instance test pattern.
     @visibleForTesting
     static Future<String> Function()? testEnsureCopiedOverride;

     /// Idempotent. Returns the absolute filesystem path to the copied archive.
     /// Throws [FileSystemException] on copy failure (gate screen routes to /error).
     static Future<String> ensureCopied() async {
       final override = testEnsureCopiedOverride;
       if (override != null) return override();

       try {
         final supportDir = await getApplicationSupportDirectory();
         final mapsDir = Directory(p.join(supportDir.path, kPmtilesMapsSubdir));
         if (!await mapsDir.exists()) {
           await mapsDir.create(recursive: true);
         }
         final dstPath = p.join(mapsDir.path, kPmtilesBasename);
         final dst = File(dstPath);

         final bundled = await rootBundle.load(kPmtilesAssetPath);
         final bundledBytes = bundled.lengthInBytes;

         if (await dst.exists() && dst.lengthSync() == bundledBytes) {
           // Idempotent skip — silent per CONTEXT mandate (no log on subsequent launches).
           return dstPath;
         }

         final stopwatch = Stopwatch()..start();
         await dst.writeAsBytes(bundled.buffer.asUint8List(), flush: true);
         stopwatch.stop();
         final megabytes = (bundledBytes / (1024 * 1024)).toStringAsFixed(1);
         _log.info('Copied $kPmtilesBasename (~$megabytes MB) in ${stopwatch.elapsedMilliseconds} ms');
         return dstPath;
       } on FileSystemException catch (e, st) {
         _log.severe('PMTiles copy failed: ${e.message}', e, st);
         rethrow;
       } catch (e, st) {
         // Wrap any other I/O-ish error so the caller's catch-on-FileSystemException pattern still fires.
         _log.severe('PMTiles copy failed (wrapped): $e', e, st);
         throw FileSystemException(e.toString());
       }
     }
   }
   ```

   /// Common path between in-app grant + AppLifecycleState.resumed re-check.
   /// Awaits the PMTiles copy (Pitfall 6 — map screen must mount with file
   /// on disk), routes to /error on FileSystemException with the underlying
   /// message in `extra`, otherwise routes to /map.
   ///
   /// Both paths converge here per CONTEXT.md §Integration Points: "Both
   /// the in-app prompt path AND the AppLifecycleState.resumed re-check
   /// path must hit it."
   Future<void> _ensureMapDataAndNavigate() async {
     try {
       await PmtilesAssetCopier.ensureCopied();
       if (!mounted) return;
       context.go('/map');
     } on FileSystemException catch (e) {
       if (!mounted) return;
       _log.severe('PMTiles copy failed; routing to /error', e);
       context.go('/error', extra: e.message);
     }
   }
   ```
