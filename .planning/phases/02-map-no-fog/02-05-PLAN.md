   /// Renders the bundled `Fra_Melun.pmtile` archive via flutter_map +
   /// vector_map_tiles + ProtomapsThemes.lightV3(), with a GPS-driven

   class MapScreen extends StatefulWidget {
     /// Production constructor — wires services from the router builder
     /// (uses live PMTiles path + GeolocatorService.stream).
     factory MapScreen.production({Key? key}) = _ProductionMapScreen;

     /// Test / DI constructor — accepts a [MapScreenServices] value
     /// object so tests can pump fakes (fake stream, fake on-disk PMTiles).
     const MapScreen.fromServices({super.key, required this.services});

     final MapScreenServices services;

     @override
     State<MapScreen> createState() => _MapScreenState();

   /// Production wrapper — defers service wiring until the route's
   /// builder picks it up. Allows the router to defer until after the
   /// gate screen has resolved the absolute pmtiles path. See router.dart.
   class _ProductionMapScreen extends MapScreen {
     _ProductionMapScreen({super.key})
         : super.fromServices(
             services: MapScreenServices(
               // The router's /map builder is authoritative for production
               // wiring; this constructor is unused in practice. If a
               // direct `MapScreen.production()` ever appears, it falls
               // back to in-process resolution (no Wave 2 path uses this).
               pmtilesPath: '',
               positionStreamFactory: GeolocatorService.stream,
             ),
           );
   }

   class _MapScreenState extends State<MapScreen> {
     static final Logger _log = Logger('presentation.map');

     final MapController _mapController = MapController();
     late final vector_tile_renderer.Theme _theme;             // import alias if needed; or just `Theme` — disambiguate from material's Theme
     Position? _lastFix;
     StreamSubscription<Position>? _positionSubscription;
     PmTilesVectorTileProvider? _tileProvider;

     @override
     void initState() {
       super.initState();
       _theme = ProtomapsThemes.lightV3();
       _subscribeToPositions();
       _loadTileProvider();
     }

     void _subscribeToPositions() {
       _positionSubscription = widget.services.positionStreamFactory().listen(
         (Position fix) {
           if (!mounted) return;
           setState(() => _lastFix = fix);
           _log.info('Fix: ${fix.latitude.toStringAsFixed(5)}, ${fix.longitude.toStringAsFixed(5)} ±${fix.accuracy.toStringAsFixed(0)}m');
         },
         onError: (Object e) => _log.warning('Position stream error', e),
       );
     }

     Future<void> _loadTileProvider() async {
       try {
         final provider = await PmTilesVectorTileProvider.fromSource(widget.services.pmtilesPath);
         if (!mounted) {
           // Same fire-and-forget shape as dispose: if the screen unmounted
           // while we were loading, close the just-opened archive without
           // awaiting (we're already past the build path).
           provider.archive.close();
           return;
         }
         setState(() => _tileProvider = provider);
       } catch (e, st) {
         _log.severe('Failed to open PmTiles archive', e, st);
         // Stays on the loading state — user sees a grey rect. Acceptable
         // POC fallback; real failure should already have been caught at
         // the gate screen's ensureCopied call (Plan 02-02).
       }
     }

     @override
     void dispose() {
       // Synchronous void dispose (Flutter contract: framework never awaits).
       // The Future-returning cancel/close calls are fire-and-forget — they
       // release their underlying resources as soon as the call lands; the
       // returned Future merely signals completion of any final cleanup,
       // which we don't gate on. Awaiting here would NOT make this safer
       // (the framework still wouldn't wait), it would only make the bug
       // less visible. See dispose() in <behavior> for full rationale.
       _positionSubscription?.cancel();
       _positionSubscription = null;
       _tileProvider?.archive.close();
       _tileProvider = null;
       _mapController.dispose();
       super.dispose();
     }
